  6class NeuralNetwork(abc.ABC):
  7    """
  8    An abstract class used for our internal representation of a generic NeuralNetwork. It consists of a graph of LayerNodes
  9    and a list of AlternativeRepresentations. It should be noted that this data structure it is not able
 10    to compute the input-output relation defined by the network. The properties of the computational graph are
 11    specialized in the concrete classes.
 12
 13    Attributes
 14    ----------
 15    nodes : dict <str, SingleInputLayerNode>
 16        Dictionary containing str keys and LayerNodes values. It contains the nodes of the graph,
 17        the identifier of the node of interest is used as a key in the nodes' dictionary.
 18    edges : dict <str, list <str>>
 19        Dictionary of identifiers of LayerNodes, it contains for each node identified by the keys, the list of nodes
 20        connected to it.
 21    alt_rep_cache : List<AlternativeRepresentation>
 22        Dictionary of containing str keys and AlternativeRepresentation values, it contains the
 23        AlternativeRepresentations of out network.
 24    up_to_date : bool
 25        Boolean Flag which shows if the internal representation is up-to-date with respect to the
 26        AlternativeRepresentations.
 27    identifier : str
 28        Identifier of the Sequential Neural Network.
 29
 30    """
 31
 32    def __init__(self, identifier: str = ''):
 33        self.nodes = {}
 34        self.edges = {}
 35        self.alt_rep_cache = []
 36        self.up_to_date = True
 37        self.identifier = identifier
